<script>
    const canvas = document.getElementById('game-canvas');
    const ctx = canvas.getContext('2d');
    const overlay = document.getElementById('overlay');
    const msgTitle = document.getElementById('msg-title');
    const msgBody = document.getElementById('msg-body');
    const levelDisplay = document.getElementById('level-display');
    const scoreDisplay = document.getElementById('score-display');

    function resize() {
        const size = Math.min(window.innerWidth - 40, window.innerHeight - 280);
        canvas.width = size;
        canvas.height = size;
    }
    window.addEventListener('resize', resize);
    resize();

    // Game Variables
    let gameActive = true;
    let level = 1;
    let score = 0;
    let state = 'idle'; // idle, dropping, lifting, caught
    let craneX = canvas.width / 2;
    let craneY = 50;
    let craneSpeed = 6;
    let dropSpeed = 5;
    let liftSpeed = 4;
    let clawOpen = true;

    // Moving Gem
    const gem = {
        x: 0,
        y: 0,
        size: 45,
        speed: 2,
        direction: 1
    };

    // Obstacles
    const obstacles = [];

    function initLevel() {
        gem.x = canvas.width / 2;
        gem.y = canvas.height - 60;
        gem.speed = 1.5 + (level * 0.5); // Gem moves faster each level
        
        obstacles.length = 0;
        const count = Math.min(3 + level, 7); // More clouds each level
        for (let i = 0; i < count; i++) {
            obstacles.push({
                x: Math.random() * (canvas.width - 80),
                y: 120 + i * ( (canvas.height - 250) / count),
                width: 80,
                height: 50,
                speed: (1.5 + Math.random() * 2 + (level * 0.3)) * (i % 2 === 0 ? 1 : -1),
                emoji: '‚òÅÔ∏è'
            });
        }
        levelDisplay.innerText = level;
        scoreDisplay.innerText = score;
    }

    initLevel();

    // Controls
    let moveLeft = false;
    let moveRight = false;
    const setupBtn = (id, start, end) => {
        const el = document.getElementById(id);
        el.addEventListener('touchstart', (e) => { e.preventDefault(); start(); });
        el.addEventListener('touchend', end);
        el.addEventListener('mousedown', start);
        el.addEventListener('mouseup', end);
        el.addEventListener('mouseleave', end);
    };

    setupBtn('btn-left', () => moveLeft = true, () => moveLeft = false);
    setupBtn('btn-right', () => moveRight = true, () => moveRight = false);
    document.getElementById('btn-drop').addEventListener('click', () => {
        if (state === 'idle') state = 'dropping';
    });
    document.getElementById('btn-restart').addEventListener('click', () => {
        if (msgTitle.innerText === "GAME OVER") {
            level = 1;
            score = 0;
        } else {
            level++;
        }
        resetRound();
    });

    function showOverlay(title, body, btnText) {
        msgTitle.innerText = title;
        msgBody.innerText = body;
        document.getElementById('btn-restart').innerText = btnText;
        overlay.style.display = 'flex';
        gameActive = false;
    }

    function resetRound() {
        overlay.style.display = 'none';
        state = 'idle';
        craneX = canvas.width / 2;
        craneY = 50;
        clawOpen = true;
        gameActive = true;
        initLevel();
    }

    function update() {
        if (!gameActive) return;

        // Gem Movement (Difficult part!)
        if (state !== 'caught') {
            gem.x += gem.speed * gem.direction;
            if (gem.x > canvas.width - 30 || gem.x < 30) {
                gem.direction *= -1;
            }
        }

        // Crane Logic
        if (state === 'idle') {
            if (moveLeft) craneX -= craneSpeed;
            if (moveRight) craneX += craneSpeed;
            craneX = Math.max(40, Math.min(canvas.width - 40, craneX));
        }

        if (state === 'dropping') {
            craneY += dropSpeed;
            
            // Obstacle collision
            obstacles.forEach(obs => {
                if (craneX > obs.x && craneX < obs.x + obs.width &&
                    craneY > obs.y && craneY < obs.y + obs.height) {
                    state = 'lifting'; // Fail
                }
            });

            // Gem Catch
            const dist = Math.sqrt((craneX - gem.x)**2 + (craneY - gem.y)**2);
            if (dist < 35) {
                state = 'caught';
                clawOpen = false;
            }

            if (craneY > canvas.height - 70) state = 'lifting';
        }

        if (state === 'caught') {
            craneY -= liftSpeed;
            gem.x = craneX;
            gem.y = craneY + 25;
            if (craneY <= 50) {
                score += 100 * level;
                showOverlay("‚ú® LEVEL CLEAR! ‚ú®", `„É¨„Éô„É´ ${level} „Çí „ÇØ„É™„Ç¢„Åó„Åü„ÇàÔºÅ\n„Å§„Åé„ÅØ „ÇÇ„Å£„Å® „ÅØ„ÇÑ„Åè„Å™„Çã„ÇàÔºÅ`, "„Å§„Åé„Å∏ÔºÅ");
            }
        }

        if (state === 'lifting') {
            craneY -= liftSpeed;
            if (craneY <= 50) {
                craneY = 50;
                state = 'idle';
            }
        }

        // Obstacles move
        obstacles.forEach(obs => {
            obs.x += obs.speed;
            if (obs.x > canvas.width) obs.x = -obs.width;
            if (obs.x < -obs.width) obs.x = canvas.width;
        });
    }

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // Gem
        ctx.font = gem.size + "px serif";
        ctx.textAlign = "center";
        ctx.textBaseline = "middle";
        ctx.shadowBlur = 15;
        ctx.shadowColor = "#00ffff";
        ctx.fillText("üíé", gem.x, gem.y);
        ctx.shadowBlur = 0;

        // Clouds
        ctx.font = "55px serif";
        obstacles.forEach(obs => {
            ctx.fillText("‚òÅÔ∏è", obs.x + obs.width/2, obs.y + obs.height/2);
        });

        // Line
        ctx.strokeStyle = "#ddd";
        ctx.lineWidth = 3;
        ctx.beginPath();
        ctx.moveTo(craneX, 0);
        ctx.lineTo(craneX, craneY);
        ctx.stroke();

        // Claw
        ctx.font = "65px serif";
        ctx.fillText(clawOpen ? "üñêÔ∏è" : "‚úä", craneX, craneY + 15);
    }

    function loop() {
        update();
        draw();
        requestAnimationFrame(loop);
    }
    loop();
</script>
